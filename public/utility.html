Utility Classes

Display


.d-block {
  display: block;
}

.d-inline {
  display: inline;
}


// Fix for Firefox bug: an image styled `max-width: 100%` within an inline-block will display at its default size, and not limit its width to 100% of an ancestral container.
.d-inlineBlock {
  display: inline-block;
  max-width: 100%;
}


.d-hidden {
  display: none;
}


.d-table {
  display: table;
}
.d-tableCell {
  display: table-cell;
}
.d-tableRow {
  display: table-row;
}


// Completely remove from the flow but leave available to screen readers.
.d-hiddenVisually {
  border: 0;
  clip: rect(1px, 1px, 1px, 1px);
  height: 1px;
  overflow: hidden;
  padding: 0;
  position: absolute;
  width: 1px;
}

Position


.pos-fit,
.pos-absolute-center,
.pos-absolute{
  position: absolute;
}


// Element will be centered to its nearest relatively-positioned ancestor.
.pos-fixed-center,
.pos-absolute-center{
  left: 50%;
  top: 50%;
  transform: translate(-50%, -50%);
}

.pos-fit,
.pos-full-screen{
  bottom: 0;
  left: 0;
  margin: auto;
  right: 0;
  top: 0;
}


.pos-full-screen,
.pos-fixed-center,
.pos-fixed{
  // Make sure fixed elements are promoted into a new layer, for performance reasons.
  backface-visibility: hidden;
  position: fixed;
}
.pos-fixed-top-left{
  top: 0;
  left: 0;
}
.pos-fixed-top-right{
  top: 0;
  right: 0;
}
.pos-fixed-bottom-right{
  bottom: 0;
  right: 0;
}
.pos-fixed-bottom-left{
  bottom: 0;
  left: 0;
}


.pos-relative{
  position: relative;
}
.pos-static{

}
